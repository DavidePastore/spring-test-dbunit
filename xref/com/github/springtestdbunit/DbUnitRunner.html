40  /**
41   * Internal delegate class used to run tests with support for {@link DatabaseSetup &#064;DatabaseSetup},
42   * {@link DatabaseTearDown &#064;DatabaseTearDown} and {@link ExpectedDatabase &#064;ExpectedDatabase} annotations.
43   * 
44   * @author Phillip Webb
45   * @author Mario Zagar
46   */
47  class DbUnitRunner {
48  
49  	private static final Log logger = LogFactory.getLog(DbUnitTestExecutionListener.class);
50  
51  	private static Map<DatabaseOperation, org.dbunit.operation.DatabaseOperation> OPERATION_LOOKUP;
52  	static {
53  		OPERATION_LOOKUP = new HashMap<DatabaseOperation, org.dbunit.operation.DatabaseOperation>();
54  		OPERATION_LOOKUP.put(DatabaseOperation.UPDATE, org.dbunit.operation.DatabaseOperation.UPDATE);
55  		OPERATION_LOOKUP.put(DatabaseOperation.INSERT, org.dbunit.operation.DatabaseOperation.INSERT);
56  		OPERATION_LOOKUP.put(DatabaseOperation.REFRESH, org.dbunit.operation.DatabaseOperation.REFRESH);
57  		OPERATION_LOOKUP.put(DatabaseOperation.DELETE, org.dbunit.operation.DatabaseOperation.DELETE);
58  		OPERATION_LOOKUP.put(DatabaseOperation.DELETE_ALL, org.dbunit.operation.DatabaseOperation.DELETE_ALL);
59  		OPERATION_LOOKUP.put(DatabaseOperation.TRUNCATE_TABLE, org.dbunit.operation.DatabaseOperation.TRUNCATE_TABLE);
60  		OPERATION_LOOKUP.put(DatabaseOperation.CLEAN_INSERT, org.dbunit.operation.DatabaseOperation.CLEAN_INSERT);
61  	}

63  	/**
64  	 * Called before a test method is executed to perform any database setup.
65  	 * @param testContext The test context
66  	 * @throws Exception
67  	 */
68  	public void beforeTestMethod(DbUnitTestContext testContext) throws Exception {
69  		Collection<DatabaseSetup> annotations = getAnnotations(testContext, DatabaseSetup.class);
70  		setupOrTeardown(testContext, true, AnnotationAttributes.get(annotations));
71  	}
72  
73  	/**
74  	 * Called after a test method is executed to perform any database teardown and to check expected results.
75  	 * @param testContext The test context
76  	 * @throws Exception
77  	 */
78  	public void afterTestMethod(DbUnitTestContext testContext) throws Exception {
79  		try {
80  			verifyExpected(testContext, getAnnotations(testContext, ExpectedDatabase.class));
81  			Collection<DatabaseTearDown> annotations = getAnnotations(testContext, DatabaseTearDown.class);
82  			try {
83  				setupOrTeardown(testContext, false, AnnotationAttributes.get(annotations));
84  			} catch (RuntimeException e) {
85  				if (testContext.getTestException() == null) {
86  					throw e;
87  				}
88  				if (logger.isWarnEnabled()) {
89  					logger.warn("Unable to throw database cleanup exception due to existing test error", e);
90  				}
91  			}
92  		} finally {
93  			testContext.getConnection().close();
94  		}
95  	}

97  	private <T extends Annotation> Collection<T> getAnnotations(DbUnitTestContext testContext, Class<T> annotationType) {
98  		List<T> annotations = new ArrayList<T>();
99  		addAnnotationToList(annotations, AnnotationUtils.findAnnotation(testContext.getTestClass(), annotationType));
100 		addAnnotationToList(annotations, AnnotationUtils.findAnnotation(testContext.getTestMethod(), annotationType));
101 		return annotations;
102 	}
103 
104 	private <T extends Annotation> void addAnnotationToList(List<T> annotations, T annotation) {
105 		if (annotation != null) {
106 			annotations.add(annotation);
107 		}
108 	}

110 	private void verifyExpected(DbUnitTestContext testContext, Collection<ExpectedDatabase> annotations)
111 			throws Exception {
112 		if (testContext.getTestException() != null) {
113 			if (logger.isDebugEnabled()) {
114 				logger.debug("Skipping @DatabaseTest expectation due to test exception "
115 						+ testContext.getTestException().getClass());
116 			}
117 			return;
118 		}
119 		IDatabaseConnection connection = testContext.getConnection();
120 		IDataSet actualDataSet = connection.createDataSet();
121 		for (ExpectedDatabase annotation : annotations) {
122 			IDataSet expectedDataSet = loadDataset(testContext, annotation.value());
123 			if (expectedDataSet != null) {
124 				if (logger.isDebugEnabled()) {
125 					logger.debug("Veriftying @DatabaseTest expectation using " + annotation.value());
126 				}
127 				DatabaseAssertion assertion = annotation.assertionMode().getDatabaseAssertion();
128 				assertion.assertEquals(expectedDataSet, actualDataSet);
129 			}
130 		}
131 	}

133 	private IDataSet loadDataset(DbUnitTestContext testContext, String dataSetLocation) throws Exception {
134 		DataSetLoader dataSetLoader = testContext.getDataSetLoader();
135 		if (StringUtils.hasLength(dataSetLocation)) {
136 			IDataSet dataSet = dataSetLoader.loadDataSet(testContext.getTestClass(), dataSetLocation);
137 			Assert.notNull(dataSet,
138 					"Unable to load dataset from \"" + dataSetLocation + "\" using " + dataSetLoader.getClass());
139 			return dataSet;
140 		}
141 		return null;
142 	}

144 	private void setupOrTeardown(DbUnitTestContext testContext, boolean isSetup,
145 			Collection<AnnotationAttributes> annotations) throws Exception {
146 		IDatabaseConnection connection = testContext.getConnection();
147 		DatabaseOperation lastOperation = null;
148 		for (AnnotationAttributes annotation : annotations) {
149 			for (String dataSetLocation : annotation.getValue()) {
150 				DatabaseOperation operation = annotation.getType();
151 				org.dbunit.operation.DatabaseOperation dbUnitDatabaseOperation = getDbUnitDatabaseOperation(operation,
152 						lastOperation);
153 				IDataSet dataSet = loadDataset(testContext, dataSetLocation);
154 				if (dataSet != null) {
155 					if (logger.isDebugEnabled()) {
156 						logger.debug("Executing " + (isSetup ? "Setup" : "Teardown") + " of @DatabaseTest using "
157 								+ operation + " on " + dataSetLocation);
158 					}
159 					dbUnitDatabaseOperation.execute(connection, dataSet);
160 					lastOperation = operation;
161 				}
162 			}
163 		}
164 	}
165 
166 	private org.dbunit.operation.DatabaseOperation getDbUnitDatabaseOperation(DatabaseOperation operation,
167 			DatabaseOperation lastOperation) {
168 		if ((operation == DatabaseOperation.CLEAN_INSERT) && (lastOperation == DatabaseOperation.CLEAN_INSERT)) {
169 			operation = DatabaseOperation.INSERT;
170 		}
171 		org.dbunit.operation.DatabaseOperation databaseOperation = OPERATION_LOOKUP.get(operation);
172 		Assert.state(databaseOperation != null, "The databse operation " + operation + " is not supported");
173 		return databaseOperation;
174 	}

176 	private static class AnnotationAttributes {
177 
178 		private DatabaseOperation type;
179 
180 		private String[] value;
181 
182 		public AnnotationAttributes(Annotation annotation) {
183 			Assert.state((annotation instanceof DatabaseSetup) || (annotation instanceof DatabaseTearDown),
184 					"Only DatabaseSetup and DatabaseTearDown annotations are supported");
185 			Map<String, Object> attributes = AnnotationUtils.getAnnotationAttributes(annotation);
186 			this.type = (DatabaseOperation) attributes.get("type");
187 			this.value = (String[]) attributes.get("value");
188 		}
189 
190 		public DatabaseOperation getType() {
191 			return this.type;
192 		}
193 
194 		public String[] getValue() {
195 			return this.value;
196 		}
197 
198 		public static <T extends Annotation> Collection<AnnotationAttributes> get(Collection<T> annotations) {
199 			List<AnnotationAttributes> annotationAttributes = new ArrayList<AnnotationAttributes>();
200 			for (T annotation : annotations) {
201 				annotationAttributes.add(new AnnotationAttributes(annotation));
202 			}
203 			return annotationAttributes;
204 		}
205 	}
