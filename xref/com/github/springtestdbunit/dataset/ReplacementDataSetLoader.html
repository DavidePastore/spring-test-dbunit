11  /**
12   * A {@link DataSetLoader data set loader} that uses a {@link ReplacementDataSet} to replace specific objects or
13   * sub-strings. By default will replace "[null]" with <code>null</code>.
14   *
15   * @author Stijn Van Bael
16   * @author Phillip Webb
17   */
18  public class ReplacementDataSetLoader implements DataSetLoader {
19  
20  	public static final Map<String, Object> DEFAULT_OBJECT_REPLACEMENTS = Collections.singletonMap("[null]", null);
21  
22  	private final DataSetLoader dataSetLoader;
23  
24  	private final Map<Object, Object> objectReplacements;
25  
26  	private final Map<String, String> subStringReplacements;
27  
28  	/**
29  	 * Create a new {@link ReplacementDataSetLoader} using a {@link FlatXmlDataSetLoader} to load the source data and
30  	 * with {@link #DEFAULT_OBJECT_REPLACEMENTS}.
31  	 */
32  	public ReplacementDataSetLoader() {
33  		this(new FlatXmlDataSetLoader());
34  	}

36  	/**

38  	 * @param dataSetLoader the source data set loader
39  	 */

41  		this(dataSetLoader, DEFAULT_OBJECT_REPLACEMENTS);
42  	}

44  	/**

47  	 * @param objectReplacements the object replacements or {@code null} if no object replacements are required
48  	 */
49  	public ReplacementDataSetLoader(DataSetLoader dataSetLoader, Map<?, ?> objectReplacements) {
50  		this(dataSetLoader, objectReplacements, null);
51  	}
52  
53  	/**
54  	 * Create a new {@link ReplacementDataSetLoader}.
55  	 * @param dataSetLoader the source data set loader
56  	 * @param objectReplacements the object replacements or {@code null} if no object replacements are required
57  	 * @param subStringReplacements the sub-string replacements or {@code null} if no sub-string replacements are
58  	 * required
59  	 */
60  	public ReplacementDataSetLoader(DataSetLoader dataSetLoader, Map<?, ?> objectReplacements,
61  			Map<String, String> subStringReplacements) {
62  		Assert.notNull(dataSetLoader, "Delegate must not be null");
63  		this.dataSetLoader = dataSetLoader;
64  		this.objectReplacements = unmodifiableMap(objectReplacements);
65  		this.subStringReplacements = unmodifiableMap(subStringReplacements);
66  	}
67  
68  	private <K, V> Map<K, V> unmodifiableMap(Map<? extends K, ? extends V> map) {
69  		Map<K, V> result = new LinkedHashMap<K, V>();
70  		if (map != null) {
71  			result.putAll(map);
72  		}
73  		return Collections.unmodifiableMap(result);
74  	}
75  
76  	public IDataSet loadDataSet(Class<?> testClass, String location) throws Exception {
77  		IDataSet dataSet = this.dataSetLoader.loadDataSet(testClass, location);
78  		ReplacementDataSet replacementDataSet = new ReplacementDataSet(dataSet, this.objectReplacements,
79  				this.subStringReplacements);
80  		return replacementDataSet;
81  	}
