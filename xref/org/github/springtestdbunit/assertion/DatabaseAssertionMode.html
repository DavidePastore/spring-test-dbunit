16  package org.github.springtestdbunit.assertion;
17  
18  import org.github.springtestdbunit.annotation.ExpectedDatabase;
19  
20  /**
21   * Database assertion modes which determine {@link ExpectedDatabase} behaviour.
22   * 
23   * @author Mario Zagar
24   */
25  public enum DatabaseAssertionMode {
26  
27  	/**
28  	 * Will use default DbUnit data sets assertions.
29  	 */
30  	DEFAULT(new DefaultDatabaseAssertion()),
31  
32  	/**
33  	 * Allows specifying only specific columns and tables in expected data set. Unspecified tables and columns are
34  	 * ignored. </p> <strong>Notes:</strong>
35  	 * <ul>
36  	 * <li>Expected row order must match order in actual data set.</li>
37  	 * <li>Specified columns must match in all rows, e.g. specifying 'column1' value without 'column2' value in one row
38  	 * and only 'column2' value in another is not allowed - both 'column1' and 'column2' values must be specified in all
39  	 * rows.</li>
40  	 * </ul>
41  	 */
42  	NON_STRICT(new NonStrictDatabaseAssertion());
43  
44  	private DatabaseAssertion databaseAssertion;
45  
46  	private DatabaseAssertionMode(DatabaseAssertion databaseAssertion) {
47  		this.databaseAssertion = databaseAssertion;
48  	}
49  
50  	public DatabaseAssertion getDatabaseAssertion() {
51  		return this.databaseAssertion;
52  	}
53  }
